## §6 How to Use This System

### Daily Operations (When we begin execution)

```bash
# Check what needs attention
sqlite3 data/workbench/workbench.db "SELECT * FROM v_project_summary ORDER BY priority;"

# See all P0 tasks
sqlite3 data/workbench/workbench.db "SELECT id, title FROM work_items WHERE priority='P0' AND status='backlog' ORDER BY workstream;"

# See unmined artifacts ready for extraction
sqlite3 data/workbench/workbench.db "SELECT name, partition, path FROM artifacts WHERE mining_status='unmined' AND sovereignty_score >= 8;"

# Query decisions
sqlite3 data/workbench/workbench.db "SELECT * FROM v_recent_decisions;"

# Pick a task by workstream
sqlite3 data/workbench/workbench.db "SELECT id, title FROM work_items WHERE workstream='F' AND status='backlog' ORDER BY id;"

# View full mining pipeline
sqlite3 data/workbench/workbench.db "SELECT * FROM v_mining_pipeline;"
```

### Querying Assets by Value

```bash
# Show all P0 artifacts (mine first)
sqlite3 data/workbench/workbench.db "
SELECT name, substr(partition, 1, 10) AS part, sovereignty_score, mining_status
FROM artifacts
WHERE sovereignty_score >= 8
ORDER BY sovereignty_score DESC, effort_to_extract;
"

# Show work items by project
sqlite3 data/workbench/workbench.db "
SELECT p.name, COUNT(w.id) AS total,
  SUM(CASE WHEN w.status='done' THEN 1 ELSE 0 END) AS done
FROM projects p
LEFT JOIN work_items w ON w.project_id = p.id
GROUP BY p.id
ORDER BY p.priority;
"

# Show decisions grouped by project
sqlite3 data/workbench/workbench.db "
SELECT p.name, COUNT(d.id) AS decisions
FROM projects p
LEFT JOIN decisions d ON d.project_id = p.id
GROUP BY p.id
ORDER BY decisions DESC;
"
```
